The following objects are available from the top-level export:

```js
import {
  ControlsManager,
  RAYCASTER_EVENTS,
  ANCHOR_MODE,
  DEFAULT_HANDLE_GROUP_NAME,
  Translation,
  Rotation,
  Pick,
  PickPlane,
  PickPlaneGroup,
  RotationGroup,
  TranslationGroup
} from 'three-freeform-controls';
```
